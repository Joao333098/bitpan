Realtime API

Connect via WebSocket

Below is a working example that connects to the Nova 2 Sonic model, sets the voice and system instructions, sends a text message, receives a spoken audio response and saves it to a wav file.

#!/usr/bin/env python3
"""
Usage:
    export NOVA_API_KEY=your_key
    python nova_client.py
"""

import asyncio
import base64
import json
import ssl
import wave
import array
import math
import os

try:
    import websockets
except ImportError:
    print("Install: pip install websockets")
    exit(1)

SAMPLE_RATE = 24000
CHANNELS = 1
SAMPLE_WIDTH = 2

def generate_silence(duration_ms: int) -> bytes:
    """Generate silence"""
    samples = (SAMPLE_RATE * duration_ms) // 1000
    return bytes(samples * SAMPLE_WIDTH)

async def main():
    api_key = os.getenv("NOVA_API_KEY")
    if not api_key:
        print("Error: Set NOVA_API_KEY environment variable")
        exit(1)
    
    ssl_context = ssl.create_default_context()
    ssl_context.check_hostname = False
    ssl_context.verify_mode = ssl.CERT_NONE
    
    url = "wss://api.nova.amazon.com/v1/realtime?model=nova-2-sonic-v1"
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Origin": "https://api.nova.amazon.com"
    }
    
    print("Connecting to Nova API...")
    
    async with websockets.connect(url, ssl=ssl_context, additional_headers=headers) as ws:
        print("✓ Connected")
        
        # Wait for session.created
        event = json.loads(await ws.recv())
        assert event["type"] == "session.created"
        print(f"← {event['type']}")
        
        # Configure session
        await ws.send(json.dumps({
            "type": "session.update",
            "session": {
                "type": "realtime",
                "instructions": "Respond briefly to greetings.",
                "audio": {
                    "input": {
                        "turn_detection": {
                            "threshold": 0.5
                        }
                    },
                    "output": {
                        "voice": "olivia"
                    }
                }
            }
        }))
        
        # Wait for session.updated
        event = json.loads(await ws.recv())
        assert event["type"] == "session.updated"
        print(f"← {event['type']}")
        
        # Start continuous silence task (like test does)
        async def send_silence():
            silence = base64.b64encode(generate_silence(100)).decode('utf-8')
            while True:
                try:
                    await ws.send(json.dumps({
                        "type": "input_audio_buffer.append",
                        "audio": silence
                    }))
                    await asyncio.sleep(0.1)
                except:
                    break
        
        silence_task = asyncio.create_task(send_silence())
        print("✓ Silence stream started")
        
        # Send text message instead of audio (more reliable)
        print("\nSending text message: 'hello how are you'")
        await ws.send(json.dumps({
            "type": "conversation.item.create",
            "item": {
                "type": "message",
                "role": "user",
                "content": [
                    {
                        "type": "input_text",
                        "text": "hello how are you"
                    }
                ]
            }
        }))
        
        # Collect response
        response_audio = bytearray()
        timeout = 15.0
        start = asyncio.get_event_loop().time()
        
        print("\nWaiting for response...")
        while asyncio.get_event_loop().time() - start < timeout:
            try:
                event = json.loads(await asyncio.wait_for(ws.recv(), timeout=2.0))
                event_type = event["type"]
                
                print(f"← {event_type}")
                
                if event_type == "error":
                    print(f"❌ ERROR: {json.dumps(event.get('error', {}), indent=2)}")
                    break
                elif event_type == "conversation.item.input_audio_transcription.completed":
                    print(f"   User: {event.get('transcript', '')}")
                elif event_type == "response.output_audio.delta":
                    response_audio.extend(base64.b64decode(event["delta"]))
                elif event_type == "response.output_audio_transcript.done":
                    print(f"   Assistant: {event.get('transcript', '')}")
                elif event_type == "response.done":
                    break
            except asyncio.TimeoutError:
                print("⏱️  Waiting...")
            except Exception as e:
                print(f"❌ Exception: {e}")
                break
        
        silence_task.cancel()
        
        # Save audio
        if response_audio:
            with wave.open("nova_response.wav", 'wb') as wav:
                wav.setnchannels(CHANNELS)
                wav.setsampwidth(SAMPLE_WIDTH)
                wav.setframerate(SAMPLE_RATE)
                wav.writeframes(bytes(response_audio))
            print(f"\n✅ Saved nova_response.wav ({len(response_audio)} bytes)")
        else:
            print("\n⚠️  No audio received")

if __name__ == "__main__":
    asyncio.run(main())

import base64
import json
import struct
import soundfile as sf
from websocket import create_connection

# ... create websocket-client named ws ...

def float_to_16bit_pcm(float32_array):
    clipped = [max(-1.0, min(1.0, x)) for x in float32_array]
    pcm16 = b''.join(struct.pack('<h', int(x * 32767)) for x in clipped)
    return pcm16

def base64_encode_audio(float32_array):
    pcm_bytes = float_to_16bit_pcm(float32_array)
    encoded = base64.b64encode(pcm_bytes).decode('ascii')
    return encoded

files = [
    './path/to/sample1.wav',
    './path/to/sample2.wav',
    './path/to/sample3.wav'
]

for filename in files:
    data, samplerate = sf.read(filename, dtype='float32')
    channel_data = data[:, 0] if data.ndim > 1 else data
    base64_chunk = base64_encode_audio(channel_data)

    # Send the client event
    event = {
        "type": "input_audio_buffer.append",
        "audio": base64_chunk
    }
    ws.send(json.dumps(event))
